RUTAS:
    /PRODUCTOS:
     GET - http://localhost:3001/categories                      (retorna todas las categor√≠as)
     GET - http://localhost:3001/products                        (retorna todos los productos)
     GET - http://localhost:3001/products?name=nombre%20del%20producto (retorna detalle del producto) - [Nota: en los espacios entre palabras se debe agregar: %20]
     GET - http://localhost:3001/product/id                      (retorna detalle del producto)
     GET - http://localhost:3001/product/reviews/id              (retorna todas las reviews del producto)

     POST - http://localhost:3001/create/product                 (crea un nuevo producto) 
     PUT - http://localhost:3001/update/product                  (modifica un producto)
     DELETE - http://localhost:3001/delete/product               (borra un producto)



    /CLIENTES:
     GET - http://localhost:3001/clients                        (retorna todos los clientes)
     GET - http://localhost:3001/clients/id                     (retorna datos del cliente)
     GET - http://localhost:3001/clients/reviews                (retorna todas las reviews del cliente)
     
     POST - http://localhost:3001/new/client                    (crea un nuevo cliente)
     PUT - http://localhost:3001/update/client                  (modifica los datos de un cliente)
     DELETE - http://localhost:3001/update/client               (elimina a un cliente)



    /ADMINISTRADOR
     GET - http://localhost:3001/sales                           (retorna todas las ventas hechas)
     GET - http://localhost:3001/orders                          (retorna todas las ventas pendientes)
